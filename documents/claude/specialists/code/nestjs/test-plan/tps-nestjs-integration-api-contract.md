## Patterns

### Pattern INT-C.1: Pact Consumer Test (Consumer Side)

```typescript
// test/contract/loan-service-payment-service.pact.spec.ts
import { PactV3, MatchersV3 } from '@pact-foundation/pact';

const provider = new PactV3({
  consumer: 'loan-service',
  provider: 'payment-service',
  dir: './pacts',
});

describe('Loan Service → Payment Service Contract', () => {
  it('should receive payment confirmation', async () => {
    await provider
      .given('a valid loan exists')
      .uponReceiving('a disbursement request')
      .withRequest({
        method: 'POST',
        path: '/payments/disburse',
        headers: { 'Content-Type': 'application/json' },
        body: { loanId: MatchersV3.uuid(), amount: MatchersV3.decimal(1000.00), currency: 'USD' },
      })
      .willRespondWith({
        status: 201,
        body: {
          transactionId: MatchersV3.uuid(),
          status: MatchersV3.string('COMPLETED'),
          processedAt: MatchersV3.iso8601DateTime(),
        },
      })
      .executeTest(async (mockServer) => {
        const client = new PaymentGatewayClient(mockServer.url);
        const result = await client.disburse({ loanId: 'uuid', amount: 1000, currency: 'USD' });
        expect(result.status).toBe('COMPLETED');
      });
  });
});
```

### Pattern INT-C.2: Pact Provider Verification (Provider Side)

```typescript
// test/contract/verify-payment-service.spec.ts
import { Verifier } from '@pact-foundation/pact';

describe('Payment Service Provider Verification', () => {
  let app: INestApplication;

  beforeAll(async () => {
    app = await createTestApp(); // real app with test DB
    await app.listen(0);
  });

  afterAll(async () => { await app.close(); });

  it('should satisfy loan-service contract', async () => {
    const verifier = new Verifier({
      providerBaseUrl: await app.getUrl(),
      pactUrls: ['./pacts/loan-service-payment-service.json'],
      stateHandlers: {
        'a valid loan exists': async () => {
          await seedTestData(); // setup required state
        },
      },
    });
    await verifier.verifyProvider();
  });
});
```

### Pattern INT-C.3: OpenAPI Schema Validation

```typescript
// test/contract/openapi-validation.spec.ts
import SwaggerParser from '@apidevtools/swagger-parser';
import * as request from 'supertest';

describe('OpenAPI Schema Compliance', () => {
  let app: INestApplication;
  let spec: any;

  beforeAll(async () => {
    app = await createTestApp();
    spec = await SwaggerParser.validate('./swagger.json'); // generated by @nestjs/swagger
  });

  it('should have all documented endpoints returning correct schemas', async () => {
    for (const [path, methods] of Object.entries(spec.paths)) {
      for (const [method, operation] of Object.entries(methods as any)) {
        if (['get', 'post', 'put', 'delete'].includes(method)) {
          const res = await request(app.getHttpServer())[method](path.replace(/{(\w+)}/g, 'test-id'));
          // Verify response matches documented schema
          expect([200, 201, 400, 401, 404]).toContain(res.status);
        }
      }
    }
  });

  it('should not have breaking changes from previous version', async () => {
    // Compare current spec with previous version
    const previousSpec = await SwaggerParser.validate('./swagger.previous.json');
    const breaking = detectBreakingChanges(previousSpec, spec);
    expect(breaking).toHaveLength(0);
  });
});
```

### Pattern INT-C.4: gRPC Contract Testing

```typescript
// test/contract/grpc-contract.spec.ts
describe('Risk Service gRPC Contract', () => {
  it('should match proto definition', async () => {
    const proto = loadProto('./proto/risk-service.proto');
    const client = createGrpcClient(proto, app.getUrl());

    const response = await client.assessRisk({ loanId: 'uuid', amount: 10000 });

    // Verify response matches proto message definition
    expect(response).toHaveProperty('riskScore');
    expect(response).toHaveProperty('riskLevel');
    expect(typeof response.riskScore).toBe('number');
    expect(['LOW', 'MEDIUM', 'HIGH']).toContain(response.riskLevel);
  });
});
```
