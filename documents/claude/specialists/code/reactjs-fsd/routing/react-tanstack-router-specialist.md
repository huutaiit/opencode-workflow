## Patterns

### Pattern 45.1: Router Creation (CRITICAL)

```typescript
// src/app/routes/__root.tsx
import { createRootRouteWithContext, Outlet } from '@tanstack/react-router';
import { TanStackRouterDevtools } from '@tanstack/router-devtools';
import type { QueryClient } from '@tanstack/react-query';

interface RouterContext {
  queryClient: QueryClient;
  auth: { user: User | null; isAuthenticated: boolean };
}

export const Route = createRootRouteWithContext<RouterContext>()({
  component: () => (
    <>
      <AppLayout>
        <Outlet />
      </AppLayout>
      {import.meta.env.DEV && <TanStackRouterDevtools />}
    </>
  ),
  notFoundComponent: () => <NotFoundPage />,
});
```

```typescript
// src/app/routes/routeTree.gen.ts (auto-generated by plugin)
// or manual:
import { createRouter } from '@tanstack/react-router';
import { routeTree } from './routeTree.gen';

export const router = createRouter({
  routeTree,
  context: {
    queryClient: undefined!, // Will be provided in RouterProvider
    auth: undefined!,
  },
  defaultPreload: 'intent',
  defaultPreloadStaleTime: 0,
});

// Type registration
declare module '@tanstack/react-router' {
  interface Register {
    router: typeof router;
  }
}

// src/app/index.tsx
<RouterProvider router={router} context={{ queryClient, auth: authState }} />
```

### Pattern 45.2: Type-Safe Route Definitions (CRITICAL)

```typescript
// src/app/routes/users/index.tsx
import { createFileRoute } from '@tanstack/react-router';
import { z } from 'zod';

const usersSearchSchema = z.object({
  page: z.number().int().positive().default(1),
  pageSize: z.number().int().positive().default(20),
  search: z.string().optional(),
  role: z.enum(['admin', 'manager', 'viewer']).optional(),
  sortField: z.enum(['name', 'email', 'createdAt']).default('name'),
  sortOrder: z.enum(['asc', 'desc']).default('asc'),
});

export const Route = createFileRoute('/users/')({
  validateSearch: usersSearchSchema,
  component: UsersListPage,
  loaderDeps: ({ search }) => ({ search }),
  loader: async ({ context: { queryClient }, deps: { search } }) => {
    await queryClient.ensureQueryData(userQueries.list(search));
  },
});

function UsersListPage() {
  const { page, pageSize, search, role } = Route.useSearch(); // Fully typed!
  const navigate = Route.useNavigate();

  // Navigate with type-safe search params
  const handlePageChange = (newPage: number) => {
    navigate({ search: (prev) => ({ ...prev, page: newPage }) });
  };
}
```

### Pattern 45.3: Search Params with Zod (CRITICAL)

```typescript
// Search params are validated and typed at the route level
const orderSearchSchema = z.object({
  status: z.enum(['pending', 'approved', 'shipped', 'delivered']).optional(),
  dateFrom: z.string().date().optional(),
  dateTo: z.string().date().optional(),
  customerId: z.string().uuid().optional(),
});

export const Route = createFileRoute('/orders/')({
  validateSearch: orderSearchSchema,
  component: OrdersPage,
});

function OrdersPage() {
  const search = Route.useSearch(); // { status?: ..., dateFrom?: ..., ... }
  const navigate = Route.useNavigate();

  // Type-safe navigation — compiler catches invalid params
  navigate({
    search: { status: 'pending', dateFrom: '2024-01-01' },
  });

  // navigate({ search: { status: 'invalid' } }); // ← TypeScript ERROR
}
```

### Pattern 45.4: Route Loaders (HIGH)

```typescript
// Loader runs before component renders — data is ready
export const Route = createFileRoute('/users/$userId')({
  loader: async ({ params, context: { queryClient } }) => {
    // params.userId is typed as string
    const user = await queryClient.ensureQueryData(userQueries.detail(params.userId));
    return { user };
  },
  component: UserDetailPage,
});

function UserDetailPage() {
  const { user } = Route.useLoaderData(); // Typed! user: User
  return <UserProfile user={user} />;
}
```

### Pattern 45.5: Route Context (Auth, Permissions) (HIGH)

```typescript
// Auth check in beforeLoad — runs before loader and component
export const Route = createFileRoute('/admin')({
  beforeLoad: ({ context }) => {
    if (!context.auth.isAuthenticated) {
      throw redirect({ to: '/login', search: { redirect: location.href } });
    }
    if (context.auth.user?.role !== 'admin') {
      throw redirect({ to: '/unauthorized' });
    }
  },
  component: AdminLayout,
});

// All child routes inherit the auth check
// /admin/users, /admin/settings — all protected
```

### Pattern 45.7: Migration from React Router (MEDIUM)

| React Router v7 | TanStack Router v1 |
|:---------------:|:-----------------:|
| `createBrowserRouter` | `createRouter` |
| `useParams()` | `Route.useParams()` |
| `useSearchParams()` | `Route.useSearch()` (typed!) |
| `useNavigate()` | `Route.useNavigate()` (typed!) |
| `<Outlet />` | `<Outlet />` (same) |
| `loader()` | `loader()` (similar) |
| `useLoaderData()` | `Route.useLoaderData()` (typed!) |
| `lazy()` | `lazyRouteComponent()` |

**Key advantage**: TanStack Router has end-to-end type safety — params, search params, loader data, and navigation are all inferred from route definitions.
